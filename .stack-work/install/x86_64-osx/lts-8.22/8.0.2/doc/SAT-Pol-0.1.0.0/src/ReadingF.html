varFold :: [String] -> Vect F2 (Lex String)
varFold (x:xs) | x == "c" || x == "p" = 1
varFold xs = foldl' (\acc x -> disj (var' x) acc) zerov xs

-------------------------------------------------------------------------------

var' "0"      = zerov
var' ('-':xs) = 1 + var ('x':xs)
var' x        = var ('x':x)

-------------------------------------------------------------------------------

disj :: Num a => a -> a -> a
disj x y = x + y + x*y
